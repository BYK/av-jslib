563 	this._applyFilter=function(row, colIndex)
564 	{
565 		var result=false;
566 		
567 		if (this.columnProperties[colIndex].filterFunction && this.columnProperties[colIndex].filter)
568 		{
569 			var filterStr=this.columnProperties[colIndex].filter;
570 			var invert=(filterStr.charAt(0)=='!');
571 			
572 			if (invert || filterStr.charAt(0)==' ')
573 				filterStr=filterStr.substr(1);
574 			
575 			result=this.columnProperties[colIndex].filterFunction(this._extractCellValue(row, colIndex), filterStr);
576 			
577 			if (invert)
578 				result=!result;
579 		}
580 		
581 		return result;
582 	};

584 	this._alphaNumericFilter=function(value, filter)
585 	{
586 		value=value.toLowerCase();
587 		var expression=(filter.charAt(0)=='*');
588 		
589 		if (expression || (filter.charAt(0)==' '))
590 			filter=filter.substr(1);
591 		
592 		filter=new RegExp((expression)?filter:filter.escapeRegExp(), "gi");
593 
594 		return !value.match(filter);
595 	};

597 	this._numericFilter=function(value, filter)
598 	{
599 		if (!isNaN(filter))
600 			filter='==' + filter;
601 		else if (filter.charAt(0)=='=' && !isNaN(filter.substr(1)))
602 			filter='=' + filter;
603 
604 		if (filter.match(/^([><]+=*|==)\d+\.?\d*$/))
605 			return !eval('(' + value + filter + ')');
606 		else
607 			return false;
608 	};

780 	{
781 		var currentRow=groupHeader.nextSibling;
782 		groupHeader.groupCount=0;
783 		while (currentRow && currentRow.parentRow==groupHeader)
784 		{
785 			for (var i=0; i<this.colCount; i++)
786 			{
787 				if (i==this.sortBy || this.columnProperties[i].hidden)
788 					continue;
789 				dataType=(this.columnProperties[i].dontSum)?"string":this.columnProperties[i].dataType;
790 				switch(dataType)
791 				{
792 					case 'int':
793 					case 'real':
794 						groupHeader.childNodes[i].firstChild.nodeValue=parseFloat(groupHeader.childNodes[i].firstChild.nodeValue) + parseFloat(currentRow.childNodes[i].firstChild.nodeValue);
795 						break;
796 					default:
797 						groupHeader.childNodes[i].appendChild(document.createElement('br'));
798 						if (groupHeader.childNodes[i].lastStr!=currentRow.childNodes[i].firstChild.nodeValue)
799 						{
800 							groupHeader.childNodes[i].appendChild(document.createTextNode(currentRow.childNodes[i].firstChild.nodeValue));
801 							groupHeader.childNodes[i].lastStr=currentRow.childNodes[i].firstChild.nodeValue;
802 						}
803 				}
804 			}
805 			groupHeader.groupCount++;
806 			groupHeader.parentNode.removeChild(currentRow);
807 			currentRow=groupHeader.nextSibling;
808 		}
