 11 /**
 12  * Escapes the special characters for a regular expression in the string.
 13  *
 14  * @return	{String}
 15  */
 16 String.prototype.escapeRegExp=function()
 17 {
 18 	matcher=new RegExp('\\\\|\\||\\(|\\)|\\[|\\{|\\^|\\$|\\*|\\+|\\?|\\.', 'gi');
 19 	var result;
 20 	var outText='';
 21 	var lastMatch=0;
 22 	while (result=matcher.exec(this))
 23 	{
 24 		outText+=this.substring(lastMatch, result.index);
 25 		outText+='\\' + result[0];
 26 		lastMatch=result.index+1;
 27 	}
 28 	delete matcher;
 29 	outText+=this.substr(lastMatch);
 30 	return outText;
 31 };

 33 /**
 34  * Replaces the occurences of the strings given in the fromArray with the toArray respectively.
 35  *
 36  * @return	{String}
 37  * @param	{String[]}	fromArray	The string array which holds the strings-to-be-replaced
 38  * @param	{String[]}	toArray	The string array which holds the strings-to-be-replaces-with.
 39  * If the length of this array is smaller than the fromArray, then the last item of this array is used for undefined indexes.
 40  * @param	{Boolean}	[dontEscape]	If given false the strings in the fromArray are used directly as regular expressions.
 41  *
 42  * @example
 43  * var myText="I love Visual Basic, CGI-Script and Ruby very much!";
 44  * myText.arrayReplace(["Visual Basic", "CGI-Script", "Ruby"], ["PHP", "JavaScript"]);
 45  * <br />Will give you "I love PHP, JavaScript and JavaScript very much!"
 46  */
 47 String.prototype.arrayReplace=function(fromArray, toArray, dontEscape)
 48 {
 49 	var expression='';
 50 	var replacementArray=new Object();
 51 	var maxToIndex=toArray.length-1;
 52 	
 53 	for (var i=0; i<fromArray.length; i++)
 54 	{
 55 		expression+='|' + ((dontEscape)?fromArray[i]:fromArray[i].escapeRegExp());
 56 		replacementArray[fromArray[i]]=toArray[Math.min(i, maxToIndex)];
 57 	}
 58 	
 59 	expression=expression.substr(1);
 60 	var matcher=new RegExp(expression, "gi");
 61 	
 62 	var result;
 63 	var outText='';
 64 	var lastMatch=0;
 65 	
 66 	while (result=matcher.exec(this))
 67 	{
 68 		outText+=this.substring(lastMatch, result.index);
 69 		outText+=replacementArray[result[0]];
 70 		lastMatch=result.index+result[0].length;
 71 	}
 72 	
 73 	outText+=this.substr(lastMatch);
 74 	return outText;	
 75 };

 77 /**
 78  * Counts and returns the given string's occurances.
 79  *
 80  * @return {integer}
 81  * @param	{String}	searchStr	The string whose occurenses will be counted.
 82  * @param	{Boolean}	[dontEscape]	If given false, the searchStr is used directly as a regular expression.
 83  */
 84 String.prototype.strCount=function(searchStr, dontEscape)
 85 {
 86 	var occurance=0;
 87 	if (!dontEscape)
 88 		searchStr=searchStr.escapeRegExp();
 89 	var matcher=new RegExp(searchStr, "gi");
 90 	while (matcher.exec(this))
 91 		occurance++;
 92 	return occurance;
 93 };

158 /**
159  * Strips all the HTML tags in the string.
160  * Removes only the given tags if tags is given
161  * 
162  * @param {String|Array} [tags] The tags which will be stripped out.
163  * @return {String}
164  */
165 String.prototype.stripHTML = function(tags)
166 {
167 	if (tags && tags.join)
168 		tags='(' + tags.join('|') + ')';
169 	else if (!tags)
170 		tags='[^>]+';
171 	tags='(<[\/]?' + tags + '>)';
172 	var matcher=new RegExp(tags, "gi");
173 	return this.replace(matcher, "");
