 74 				this.hideTimer=null;
 75 			}

 78 		/**
 79 		 * Adjusts the size and the position of the hint div according to the data inside it

 81 		 *
 82 		 * @method
 83 		 * @param {integer} [xPos] The x coordinate on the page where the hint box will be popped.
 84 		 * @param {integer} [yPos] The y coordinate on the page where the hint box will be popped.
 85 		 */
 86 		Visual.customHint.adjustSizeNPosition=function(xPos, yPos)	
 87 		{
 88 			if (!xPos)
 89 				xPos=this.lastXPos;
 90 			if (!yPos)
 91 				yPos=this.lastYPos;
 92 
 93 			this.style.height="auto";
 94 			this.style.width="auto";
 95 			var widthVar=Math.min(this.scrollWidth, Visual.clientWidth()/3);
 96 			this.style.width=widthVar + 'px';
 97 			var heightVar=Math.min(this.scrollHeight, Visual.clientHeight()/3);
 98 			this.style.height=heightVar + 'px';
 99 			this.style.left=((Visual.customHint.config["offsetX"] + xPos + widthVar) < (Visual.clientWidth() + Visual.scrollLeft())) ? (Visual.customHint.config["offsetX"] + xPos + 'px') : (Visual.customHint.config["offsetX"] + xPos - widthVar + 'px');
100 			this.style.top=((Visual.customHint.config["offsetY"] + yPos + heightVar) < (Visual.clientHeight() + Visual.scrollTop())) ? (Visual.customHint.config["offsetY"] + yPos + 'px') : (Visual.customHint.config["offsetY"] + yPos - heightVar + 'px');
101 			
102 			this.lastXPos=xPos;
103 			this.lastYPos=yPos;
104 		};

180 			/**
181 			 * The event handler for the document's onmousemove event to trigger the hints.
182 			 */
183 			function(e)
184 			{
185 				var element=(e.srcElement)?e.srcElement:e.target;
186 				var hint;
187 				if (element && element!=Visual.customHint && element.getAttribute)
188 				{
189 					while (element!=Visual.customHint && element.getAttribute && !element.getAttribute("hint") && element.parentNode && element.getAttribute("showParentHint")!="false")
190 						element=element.parentNode;
191 					hint=(element.getAttribute)?element.getAttribute("hint"):null;
192 					if (hint=="%self%")
193 						hint=element.innerHTML;
194 				}
195 				if (element && (hint==Visual.customHint.innerHTML || element==Visual.customHint))
196 				{
197 					Visual.customHint.clearTimer();
198 				}
199 				else if (!hint && element!=Visual.customHint && Visual.customHint.style.visibility=='visible' && !Visual.customHint.hideTimer)
200 				{
201 					Visual.customHint.onmouseout();
202 				}
203 				else if (element && hint && element!=Visual.customHint.lastElement)
204 				{
205 					Visual.customHint.pop(hint, e.clientX + Visual.scrollLeft(), e.clientY + Visual.scrollTop());
206 				}
207 				if (element && element!=Visual.customHint)
208 					Visual.customHint.lastElement=element;
209 			}
