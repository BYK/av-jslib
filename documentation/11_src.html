 69 /**
 70  * Creates a tiny little upload box which has the class name <b>uploadContainer</b>.
 71  * It asynchronously uploads selected file immediately after the user have selected it.
 72  *
 73  * @return {HTMLDivElementObject} The object reference to the created upload box div.
 74  * @param {String} titleText	The title of the upload box.
 75  * @param {String} postAdress	The adress of the server-side page, where the file will be uploaded.
 76  * @param {String} params	A "&" delimited string which contains the parameters which will be passed
 77  * to the server-side page. Just like the GET paramters of a page, the parameter values should be URIEncoded.
 78  * <br /><b>Note: </b>The last item should not be a name-value pair. Instead, it should only have the name of the file variable.
 79  * <br />An example params string will be like this: <br />
 80  * <br />var1=hello%20world&var2=hello%20again&theFileVar
 81  * @param {Function(HTMLDivElementObject, String)} [callBackFunc] The function which is called when the upload process is finished.
 82  * It should take the parent object of the uploadBox div, which is the same element what the createUploadBox function returns,
 83  * as its first paramter. The text result of the server-side page's response, which is defined in <b>postAddress</b> parameter, as its second parameter.
 84  * <br /> If this function returns false, the upload box will remain and if this function returns nothing or true the upload box will dissappear after
 85  * the POST operation is completed.
 86  * <br />The upload box will automatically dissappear if callBackFunction is not assigned also.
 87  */
 88 QuickEdit.createUploadBox=function(titleText, postAddress, params, callBackFunc)
 89 {
 90 	var containerDiv=document.createElement("div"); //create the container div, which contains the necessary elements for the upload box.
 91 	containerDiv.className="uploadContainer"; //set the div's class to uploadContainer both for CSS and identification
 92 	containerDiv.id="uplContainer" + QuickEdit.uploadBoxCount; //assign the unique upload div id
 93 	containerDiv.callBackFunc=callBackFunc; //define a new property for the containerDiv for storing the *callBackFunc*
 94 	
 95 	//start defining the onload function of the upcoming iframe in text format for compatibility with IE
 96 	var onloadFunc="var responseText=(this.contentDocument)?this.contentDocument.body.innerHTML:this.contentWindow.document.body.innerHTML;responseText=responseText.replace(/<\S[^><]*>/g, '');if(!responseText)return;var destroyContainer=true;if(this.parentNode.callBackFunc)destroyContainer=this.parentNode.callBackFunc(this.parentNode,responseText);if(destroyContainer)QuickEdit._destroyUploadBox(this.parentNode);";
 97 	
 98 	//prepare the inner visual structure of the uploadBox container div - this part might be customized
 99 	var inHTML='<div ondblclick="QuickEdit._destroyUploadBox(this.parentNode)" class="uploadTitle" style="float: left; clear: both; width: 100%">' + titleText + '</div>';
100 	
101 	//add the necessary hidden iframe code
102 	inHTML+='<iframe id="uploadIframe' + QuickEdit.uploadBoxCount + '" name="uploadIframe' + QuickEdit.uploadBoxCount + '" style="display:none" src="about:blank" onload="' + onloadFunc + '"></iframe>';
103 	
104 	//add the necessary form code to the container div. Keeping this part as is, is strongly recommended but might be customized
105 	inHTML+='<form style="float: left; width: 100%" action="' + postAddress + '" id="uploadForm' + QuickEdit.uploadBoxCount + '" method="post" enctype="multipart/form-data" target="uploadIframe' + QuickEdit.uploadBoxCount + '">';
106 	
107 	
108 	var paramList=params.split('&');
109 	for (var i=0; i<paramList.length-1; i++)
110 	{
111 		var tempArray=paramList[i].split('=');
112 		inHTML+='<input type="hidden" name="' + tempArray[0] + '" value="' + tempArray[1] + '" />';
113 	}
114 	//inHTML+='<input type="hidden" name="MAX_FILE_SIZE" value="500000" />';
115 	inHTML+='<input type="file" name="' + paramList[paramList.length-1] + '" onchange="if(this.value){this.form.submit();this.disabled=true;document.getElementById(\'uploadImg' + QuickEdit.uploadBoxCount + '\').style.display=\'block\'}" />';
116 	//place the hidden "in-progress" image at the bottom of the form, might be customized except the id part
117 	inHTML+='<center><img src="' + QuickEdit.config["uploadImgPath"] + '" id="uploadImg' + QuickEdit.uploadBoxCount + '" style="display: none; margin: 1px;" /></center>';
118 	inHTML+='</form>';
119 	//assign the dynamically generated HTML code to the container div's innerHTML property
120 	containerDiv.innerHTML=inHTML;
121 	document.body.appendChild(containerDiv); //add the container div to the document
122 	QuickEdit.uploadBoxCount++;
123 	return containerDiv; //return the created, final container div as an object
124 };

166 QuickEdit._changeImage=function(imgElement, uploadAddress, params, title)
167 {
168 	/*
169 	if (typeof(imgElement)=='string')
170 		imgElement=document.getElementById(imgElement);
171 	*/
172 	
173 	if (imgElement.editing) //if there is already an uploadBox, return false
174 		return false;
175

177 		title=QuickEdit.config["imgUploadTitle"];

197 QuickEdit._imgLoaded=function(container, responseText)
198 {
199 	if (responseText && responseText.substr(0, 5)=="path=")
200 	{
201 		//if there *is* a response text and it starts with "path=" prefix,
202 		//which indicates a successfull upload operation and gives us the new image's http adress
203 		//we will set the given adress to the image's src property,
204 		//but if they are the same(image.src and the new path), browsers will continue to use
205 		//the cache. To prevent this, we add a dummy get parameter which is actually the current
206 		//time, which forces a refresh.
207 		var now=new Date();
208 		container.callerElement.src=responseText.replace(/\s+$/,"").substring(5) + '?' + now.getTime();
209 		delete now;
210 		return true;
211 	}
212 	else
213 	{
214 		//if the responseText is not in the format we expected, raise an error and inform the user
215 		if (QuickEdit.config["useInfoBox"])
216 			Visual.infoBox.show(QuickEdit.config["imgUploadError"]);
217 		else
218 			alert(QuickEdit.config["imgUploadError"]);
219 		//alert(responseText);
220 		//enable the "file" input box again for a retry
221 		var inputAreas=container.getElementsByTagName("input");
222 		for (var i=0; i<inputAreas.length; i++)
223 			inputAreas[i].disabled=false;
224 		delete inputAreas;
225 		
226 		//reset the form and hide the "in-progress" image
227 		container.getElementsByTagName("form")[0].reset();
228 		container.getElementsByTagName("img")[0].style.display="none";
229 		return false;
230 	}
231 };

303 /**
304  * This function is used to send the newly edited element's data to the server-sde-scripting page
305  * and to set the new data if it is confirmed by the <i>action</i> page.
306  *
307  * @method
308  * @private
309  * @deprecated Used internally, should not be called directly.
310  * @param {HTMLElementObject} nameContainer
311  */
312 QuickEdit._setEditedValue=function(nameContainer)
313 {
314 	var labelObject=nameContainer.parentNode;
315 	var completedFunction=function(requestObject)
316 	{
317 		if (parseInt(requestObject.responseText)>0) //expecting "1" from the response page
318 		{
319 			var newName=nameContainer.value;
320 			labelObject.innerHTML="";
321 			labelObject.appendChild(document.createTextNode(newName));
322 			labelObject.innerHTML=labelObject.innerHTML.LBtoBR();
323 
324 			labelObject.editing=false;
325 			labelObject.onmouseout({type: "mouseout", target: labelObject});
326 		}
327 		else
328 		{
329 			if (QuickEdit.config["useInfoBox"])
330 				Visual.infoBox.show(QuickEdit.config["textEditError"]);
331 			else
332 				alert(QuickEdit.config["textEditError"]);
333 			nameContainer.disabled=false;
334 		}
335 	};
336 	var loadFunction=function()
337 	{
338 		nameContainer.disabled=true;
339 	};
340 	
341 	AJAX.makeRequest(
342 		"POST",
343 		labelObject.quickEdit.action,
344 		labelObject.quickEdit.params + '=' + encodeURIComponent(nameContainer.value.LBtoBR()),
345 		completedFunction,
346 		loadFunction
347 	);
348 };

350 /**
351  * Starts the editing processes of text based elements.
352  *
353  * @method
354  * @private
355  * @deprecated Used internally, should not be used directly.
356  * @param {HTMLElementObject} labelObject
357  */
358 QuickEdit._editLabel=function(labelObject)
359 {
360 	if (labelObject.editing || labelObject.getElementsByTagName("INPUT").length>0)
361 		return;
362 	
363 	var editBox;
364 	
365 	if (labelObject.quickEdit.type=="textarea")
366 		editBox=document.createElement("TEXTAREA");
367 	else
368 		editBox=document.createElement("INPUT");
369 
370 	editBox.style.width=(labelObject.clientWidth>79)?(labelObject.clientWidth - 4) + "px":'auto';
371 	editBox.style.height=(labelObject.scrollHeight - 4) + "px";
372 
373 	labelObject.oldHTML=labelObject.innerHTML;
374 	labelObject.innerHTML=labelObject.innerHTML.BRtoLB();
375 
376 	editBox.value=(labelObject.firstChild)?labelObject.firstChild.nodeValue:labelObject.innerText;
377 	editBox.originalValue=editBox.value;
378 
379 	labelObject.innerHTML="";
380 
381 	editBox.onblur=function() 
382 	{
383 		if (this.value==this.originalValue)
384 		{			
385 			labelObject.innerHTML=labelObject.oldHTML;
386 			labelObject.editing=false;
387 			labelObject.onmouseout({type: "mouseout", target: labelObject});
388 		}
389 		
390 		else
391 		{
392 			QuickEdit._setEditedValue(this);
393 			this.disabled=true;
394 		}
395 		
396 	};
397 	
398 	editBox.onkeydown=function(e)
399 	{
400 		var key = e ? e.which : window.event.keyCode;
401 		if (key==27)
402 		{
403 			this.value=this.originalValue;
404 			this.onblur();
405 		}
406 		else if (this.tagName=='INPUT' && key==13)
407 		{
408 			this.onblur();
409 		}
410 		else if (this.tagName=='TEXTAREA' && parseInt(this.style.height)<this.scrollHeight)
411 		{
412 			this.style.height=Math.max(this.scrollHeight - 2, 20) + "px";
413 		}
414 		
415 		return true;
416 	};
417 	
418 	labelObject.appendChild(editBox);
419 	labelObject.editing=true;
420 	
421 	editBox.className="editLabel";
422 	editBox.focus();
423 	editBox.select();
424 };
