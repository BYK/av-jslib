 20 /**
 21  * Evaluates and assigns the attributes given in attributeStr as string to the given element.
 22  * Works incrementally, which means will override existing properties but leaves the ones which
 23  * are not mentioned in attributeStr as they are.
 24  * 
 25  * @deprecated Used internally, might be used if necessary.
 26  * @return {HTMLElement} Returns the given element if succeeds, false if fails.
 27  * @param {HTMLElementObject} element The element whose attributes will be set.
 28  * @param {String} propertyName The name of the property which the parsed attributes will be assigned to.
 29  * @param {String} attributeStr The string which containts the attributes.
 30  */
 31 aParser.setElementAttributes=function(element, propertyName, attributeStr)
 32 {
 33 	var attributes;
 34 	if (!eval("attributes={" + attributeStr + "};"))
 35 		return false;
 36 	
 37 	if (!element[propertyName])
 38 		element[propertyName]=attributes;
 39 	else
 40 	{
 41 		for (var attrName in attributes)
 42 			element[propertyName][attrName]=attributes[attrName];
 43 	}
 44 	
 45 	return element;
 46 };

 48 /**
 49  * Collects the elements which satisfies the CSS query given in queryString and
 50  * assigns them the attributes given in attributeStr as text. If attributeStr
 51  * is *, then it uses the elements' inline attribute whose name is given in
 52  * propertyName to gather the element spesific attributeStr.
 53  * 
 54  * @method
 55  * @deprecated Used internally, in most cases you shouldn't be in a need for calling this function.
 56  * @param {String} queryStr The CSS query string for determination of the proper elements.
 57  * @param {String} propertyName The name of the property which the parsed attributes will be assigned to.
 58  * @param {String} attributeStr The string which containts the attributes.
 59  * @param {Function(HTMLElement)} [additionalOp] The function, which will be called for each found element.
 60  */
 61 aParser.retrieveElementsAndSetAttributes=function(queryStr, propertyName, attributeStr, additionalOp)
 62 {
 63 	var elements=cssQuery(queryStr);
 64 	
 65 	for (var i=elements.length-1; i>=0; i--)
 66 	{
 67 		if (
 68 					aParser.setElementAttributes(
 69 						elements[i],
 70 						propertyName,
 71 						(attributeStr!='*')?attributeStr:elements[i].getAttribute(propertyName)
 72 					)
 73 				)
 74 			additionalOp(elements[i]);
 75 	}	
 76 };

 78 /**
 79  * Assigns the elements' attributes rules from the ruleText
 80  * 
 81  * @method
 82  * @param {String} ruleText The text which contains the rules in an external CSS file like structure.
 83  * @param {String} propertyName The name of the property which the parsed attributes will be assigned to.
 84  * @param {Function(HTMLElement)} [additionalOp] The function, which will be called for each found element.
 85  */
 86 aParser.assignAttributesFromText=function(ruleText, propertyName, additionalOp)
 87 {
 88 	ruleText+="\n*[quickEdit]{*}";
 89 	ruleText=ruleText.replace(/\/\*.+\*\//ig, '');
 90 	var matcher=/([^\{]+)\s*\{\s*([^\}]+)\s*\}/ig;
 91 	
 92 	var result, queryStr, attributeStr;
 93 	while (result=matcher.exec(ruleText))
 94 	{
 95 		queryStr=result[1].trim();
 96 		attributeStr=result[2].trim();
 97 		
 98 		aParser.retrieveElementsAndSetAttributes(queryStr, propertyName, attributeStr, additionalOp);
 99 	}
100 };

102 /**
103  * Assigns the elements' attributes rules from the given text file.
104  * Loads the file and then calls the assignAttributesFromText to
105  * parse its text content.
106  * 
107  * @method
108  * @param {String} fileAddress The address of the file which contains the rules with a CSS file like structure.
109  * @param {String} propertyName The name of the property which the parsed attributes will be assigned to.
110  * @param {Function(HTMLElement)} [additionalOp] The function, which will be called for each found element.
111  */
112 aParser.assignAttributesFromFile=function(fileAddress, propertyName, additionalOp)
113 {
114 	AJAX.makeRequest(
115 		'GET',
116 		fileAddress,
117 		'',
118 		function(requestObject)
119 		{
120 			var ruleText='';
121 			if (requestObject.status==200 || requestObject.status==0 && requestObject.responseText)
122 				ruleText=requestObject.responseText;
123 			
124 			aParser.assignAttributesFromText(ruleText, propertyName, additionalOp);
125 		}
126 	);
127 }
