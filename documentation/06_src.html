 23 	function()

 35 			 * The time interval before the infoBox is automatically hid in milliseconds. Default is 3000.

 37 			 * @type {integer}
 38 			 */
 39 			timeout: 3000
 40 		};
 41 		
 42 		/**
 43 		 * Clears the timer which is set to hide the infoBox after some interval.
 44 		 *
 45 		 * @method
 46 		 */
 47 		Visual.infoBox.clearTimer=function()
 48 		{
 49 			if (this.hideTimer)
 50 					clearTimeout(this.hideTimer);	
 51 		}
 52 		
 53 		/**
 54 		 * Shows the info box with the given message.
 55 		 *
 56 		 * @method
 57 		 * @param {String} [message] The message which will appear in the info box.
 58 		 * The string here is assigned to the innerHTML of the infoBox object.
 59 		 * <br />If not set, the innerHTML of the infoBox is left the same.
 60 		 * @param {Boolean} [showImmediately] If set, the infoBox appears instantly, without the fade effect.
 61 		 * @param {Integer} [timeout] The message spesific timeout in milliseconds. If not given, config.timeout is used.
 62 		 */
 63 		Visual.infoBox.show=function(message, showImmediately, timeout)
 64 		{
 65 			this.clearTimer();
 66 			if (message)
 67 				this.innerHTML=message;
 68 			Visual.setOpacity(this, (showImmediately)?1:0);
 69 			this.style.visibility="visible";
 70 			
 71 			if (!timeout)
 72 				timeout=Visual.infoBox.config["timeout"];
 73 
 74 			Visual.fade(this, 1, true, function()
 75 			{
 76 				if (Visual.infoBox.config["timeout"])
 77 					this.hideTimer=setTimeout('Visual.infoBox.hide();', timeout);
 78 			}
 79 			);
 80 		};
 81 		
 82 		/**
 83 		 * Hides the infoBox with a fade effect and clears the hide timer.
 84 		 *
 85 		 * @method
 86 		 */
 87 		Visual.infoBox.hide=function()
 88 		{
 89 			this.clearTimer();
 90 			Visual.fade(this, 0, true, function() {this.style.visibility='hidden';});
 91 		};
 92
