 15 /**
 16  * Represents a namespace, Events, for creating and managing an event handler queue for any element.
 17  *
 18  * @namespace
 19  */
 20 Events = {};
 21 
 22 /**
 23  * A counter used to create unique IDs for given event handlers.
 24  *
 25  * @private
 26  * @type integer
 27  */
 28 Events.guid=1;
 29 
 30 /**
 31  * Adds the given event handler to element's the on-type event's event handler queue.
 32  *
 33  * @method
 34  * @param {Object} element The element which the event handler will be added
 35  * @param {String} type The name of the event without the "on" prefix
 36  * @param {Function(EventObject)} handler The event handler function
 37  *
 38  * @example
 39  * function resizeAlert(e)
 40  * {
 41  * 	alert("You have resized the window!");
 42  * }
 43  * Events.add(window, "resize", resizeAlert);
 44  */
 45 Events.add=function(element, type, handler)
 46 {
 47 	// assign each event handler a unique ID
 48 	if (!handler.$$guid) handler.$$guid = Events.guid++;
 49 	// create a hash table of event types for the element
 50 	if (!element.events) element.events = {};
 51 	// create a hash table of event handlers for each element/event pair
 52 	var handlers = element.events[type];
 53 	if (!handlers) {
 54 		handlers = element.events[type] = {};
 55 		// store the existing event handler (if there is one)
 56 		if (element["on" + type]) {
 57 			handlers[0] = element["on" + type];
 58 		}
 59 	}
 60 	// store the event handler in the hash table
 61 	handlers[handler.$$guid] = handler;
 62 	// assign a global event handler to do all the work
 63 	element["on" + type] = Events._handle;
 64 };
 65 
 66 /**
 67  * Removes the given event handler from element's the on-type event's event handler queue.
 68  *
 69  * @method
 70  * @param {Object} element The element which the event handler will be removed
 71  * @param {String} type The name of the event without the "on" prefix
 72  * @param {Function(EventObject)} handler The event handler function
 73  *
 74  * @example
 75  * Events.remove(window, "resize", resizeAlert);
 76  */
 77 Events.remove=function(element, type, handler)
 78 {
 79 	// delete the event handler from the hash table
 80 	if (element.events && element.events[type]) {
 81 		delete element.events[type][handler.$$guid];
 82 	}
 83 };

 85 /**
 86  * The generic event handler which manages the event queue
 87  *
 88  * @private
 89  * @param {EventObject}
 90  * @return {Boolean}
 91  */
 92 Events._handle=function(event)
 93 {
 94 	var returnValue = true;
 95 	// grab the event object (IE uses a global event object)
 96 	event = event || Events.fix(((this.ownerDocument || this.document || this).parentWindow || window).event);
 97 	if (!event._type)
 98 		event._type=event.type;
 99 	// get a reference to the hash table of event handlers
100 	var handlers = this.events[event._type];
101 	// execute each event handler
102 	for (var i in handlers) {
103 		this.$$handleEvent = handlers[i];
104 		if (this.$$handleEvent(event) === false) {
105 			returnValue = false;
106 		}
107 	}
108 	//delete this.$$handleEvent;
109 	return returnValue;
110 };

118 Events.fix=function(event)

122 	event.preventDefault = function() {this.returnValue=false;};

124 	return event;

127 /**
128  * Handles the mouse wheel event and iterprets it for a browser independent usage.
129  * <br />Adds support for the new event <b>onwheel</b> for all applicable elements.
130  *	<br /><br /><i>Main handling code from	<a href="http://adomas.org/javascript-mouse-wheel/" target="_blank">http://adomas.org/javascript-mouse-wheel/</a>
131  * <br />Implemantation and adaptation by BYK</i>
132  *
133  * @private
134  * @deprecated
135  * @return {Boolean}
136  */
137 Events._handleMouseWheelEvent=function(event)
138 {
139 	if (!event) /* For IE. */
140 		event = Events.fix(window.event);
141 
142 	event._type="wheel";
143 	event.delta=0;
144 	if (event.wheelDelta) /* IE/Opera. */
145 	{
146 		event.delta = event.wheelDelta/120;
147 		/** In Opera 9, delta differs in sign as compared to IE.
148 		 */
149 		if (window.opera)
150 			event.delta = -event.delta;
151 	}
152 	else if (event.detail) /** Mozilla case. */
153 	{
154 		/** In Mozilla, sign of delta is different than in IE.
155 		 * Also, delta is multiple of 3.
156 		 */
157 		event.delta = -event.detail/3;
158 	}
159 	/** If delta is nonzero, handle it.
160 	 * Basically, delta is now positive if wheel was scrolled up,
161 	 * and negative, if wheel was scrolled down.
162 	 */
163 	if (event.target && event.delta)
164 	{
165 		event._target=event.target;
166 		while (event._target!=document && !event._target.onwheel)
167 			event._target=event._target.parentNode;
168 		if (event._target.onwheel)
169 			event._target.onwheel(event);
170 	}
171 };
172 
173 /* Initialization code for onwheel */
174 if (window.addEventListener)
175 	// DOMMouseScroll is for mozilla
176 	window.addEventListener('DOMMouseScroll', Events._handleMouseWheelEvent, false);
177 // IE/Opera
178 window.onmousewheel = document.onmousewheel = Events._handleMouseWheelEvent;
