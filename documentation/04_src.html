 16  * Introduces a namespace that holds the sliders' functions under the Visual namespace.

 18  * @namespace

 39 /**
 40  * The event handler which responds to a slider's onmousedown event to lock the user to that slider.
 41  *
 42  * @method
 43  * @private
 44  * @deprecated Since this is an event handler, it should not be called directly.
 45  */
 46 Visual.sliders._lockSlider=function(event)
 47 {
 48 	var obj=event.target;
 49 	obj.initialPos=event.clientX;
 50 
 51 	obj.prevSib.initialWidth=parseInt(obj.prevSib.style.width);
 52 	obj.nextSib.initialWidth=parseInt(obj.nextSib.style.width);
 53 	Visual.sliders.activeSlider=obj;
 54 	if (Visual.sliders.config["effects"])
 55 		obj.className="slider_active";
 56 	return false;
 57 };

 59 /**
 60  * The event handler which responds to a slider's onmousedown event to lock the user to that slider.
 61  *
 62  * @return {Boolean}
 63  * @private
 64  * @deprecated Since this is an event handler, it should not be called directly.
 65  */
 66 Visual.sliders._unlockSlider=function(event)
 67 {
 68 	//var obj=event.target;
 69 	var obj=Visual.sliders.activeSlider;
 70 	if (!obj)
 71 		return true;
 72 	if (Visual.sliders.config["effects"])
 73 		obj.className=(event.target!=obj)?"slider":"slider_hover";
 74 	obj.initialPos=false;
 75 	if (obj.getAttribute("onSlideEnd"))
 76 		eval(obj.getAttribute("onSlideEnd") + '(obj);');
 77 	else if (obj.onSlideEnd)
 78 		obj.onSlideEnd({type: "slideEnd", target: obj});
 79 
 80 	Visual.sliders.activeSlider=false;
 81 	return Visual.sliders.activeSlider;
 82 };

 84 /**
 85  * The event handler which responds to the document's onmousemove event when a slider is active.
 86  *
 87  * @return {Boolean}
 88  * @private
 89  * @deprecated Since this is an event handler, it should not be called directly.
 90  */
 91 Visual.sliders._moveSlider=function(event)
 92 {
 93 	//var obj=event.target;
 94 	var obj=Visual.sliders.activeSlider;
 95 	if (!obj.initialPos)
 96 		return false;
 97 	var prevSibWidth=obj.prevSib.initialWidth + event.clientX - obj.initialPos;
 98 	var nextSibWidth=obj.nextSib.initialWidth - event.clientX + obj.initialPos;
 99 	if (prevSibWidth<0 || nextSibWidth<0)
100 		return false;
101 	obj.prevSib.style.width=prevSibWidth + "px";
102 	obj.nextSib.style.width=nextSibWidth + "px";
103 	obj.style.height=(Math.max(obj.prevSib.offsetHeight, obj.nextSib.offsetHeight) + heightDifference) + "px";	
104 };

106 /**
107  * The event handler which responds to a slider's ondblclick event to collapse one of the neighbour containers.
108  *
109  * @return {Boolean}
110  * @private
111  * @deprecated Since this is an event handler, it should not be called directly.
112  */
113 Visual.sliders._collapseContainer=function(event)
114 {
115 	var obj=event.target;
116 	var totalWidth=parseInt(obj.prevSib.style.width)+parseInt(obj.nextSib.style.width);
117 	var condition=(parseInt(obj.prevSib.style.width)>parseInt(obj.nextSib.style.width));
118 	
119 	Visual.fadeNSlide(
120 		(condition)?obj.prevSib:obj.nextSib,
121 		0,
122 		-1,
123 		true,
124 		true,
125 		function(o)
126 		{
127 			Visual.setOpacity(o, 1);
128 			Visual.fadeNSlide(
129 				(condition)?obj.nextSib:obj.prevSib,
130 				totalWidth,
131 				1,
132 				true,
133 				true,
134 				function (o)
135 				{
136 					if (obj.getAttribute("onSlideEnd"))
137 						eval(obj.getAttribute("onSlideEnd") + '(obj);');
138 					else if (obj.onSlideEnd)
139 						obj.onSlideEnd({type: "slideEnd", target: obj});
140 				}
141 			);
142 		}
143 	);
144 };

146 /**
147  * The initialization function which is attached to the Visual namespace's initFunctions array.
148  * Initializes the div elements whose class name is "slider" as sliders.
149  *
150  * @method
151  * @deprecated Use <a href="#Visual.init">Visual.init</a> instead
152  */
153 Visual.sliders.init=function()
154 {
155 	Visual.sliders.config["effects"]=Visual.sliders.config["effects"] && (Visual.getCSSRule(".slider_hover") && Visual.getCSSRule(".slider_active"));
156 	var divElements=document.getElementsByTagName("div");
157 	for (var i=0; i<divElements.length; i++)
158 		if (divElements[i].className=="slider")
159 		{
160 			Events.add(divElements[i], "mousedown", Visual.sliders._lockSlider);
161 			Events.add(divElements[i], "dblclick", Visual.sliders._collapseContainer);
162 			//Events.add(divElements[i], "mouseup", Visual.sliders._unlockSlider);
163 			//Events.add(divElements[i], "mousemove", Visual.sliders._moveSlider);
164 			divElements[i].onselectstart=function() {return false};
165 			//REAL previous sibling determination
166 			divElements[i].prevSib=divElements[i].previousSibling;
167 			while (divElements[i].prevSib.nodeName!="DIV")
168 				divElements[i].prevSib=divElements[i].prevSib.previousSibling;
169 			//determined and assigned the REAL previous sibling object to objects prevSib property
170 			//REAL next sibling determination
171 			divElements[i].nextSib=divElements[i].nextSibling;
172 			while (divElements[i].nextSib.nodeName!="DIV")
173 				divElements[i].nextSib=divElements[i].nextSib.nextSibling;
174 			//determined and assigned the REAL next sibling object to objects nextSib property
175 			if (Visual.sliders.config["effects"])
176 			{			
177 				Events.add(divElements[i], "mouseover", function() {if(!this.initialPos)this.className="slider_hover"});
178 				Events.add(divElements[i], "mouseout", function() {if(!this.initialPos)this.className="slider"});
179 			}
180 			
181 			divElements[i].style.height=Math.max(divElements[i].prevSib.offsetHeight, divElements[i].nextSib.offsetHeight) + "px";
182 			heightDifference=parseInt(divElements[i].style.height) - divElements[i].offsetHeight;
183 		}
184 	Visual.sliders.activeSlider=false;
185 	Events.add(document, "mousemove", Visual.sliders._moveSlider);
186 	Events.add(document, "mouseup", Visual.sliders._unlockSlider);
187 	document.onselectstart=function() {return !Visual.sliders.activeSlider};
188 	Events.add(document, "dragstart", function() {return !Visual.sliders.activeSlider});
189 };
190 
191 Visual.initFunctions.push(Visual.sliders.init);
